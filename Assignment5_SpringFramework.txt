Q1. HR Application using Spring Framework
-----------------------------------------
Steps to Solve:
1. Create a Spring Boot project.
2. Define an Employee model.
3. Create an EmployeeService and EmployeeController.
4. Run REST endpoints for basic CRUD operations.

Code:
----- Employee.java -----
package com.hrapp.model;

public class Employee {
    private int id;
    private String name;
    private String role;

    public Employee(int id, String name, String role) {
        this.id = id;
        this.name = name;
        this.role = role;
    }

    public int getId() { return id; }
    public String getName() { return name; }
    public String getRole() { return role; }
}

----- EmployeeService.java -----
package com.hrapp.service;

import java.util.*;
import org.springframework.stereotype.Service;
import com.hrapp.model.Employee;

@Service
public class EmployeeService {
    private List<Employee> employees = new ArrayList<>();

    public List<Employee> getAll() { return employees; }
    public void add(Employee e) { employees.add(e); }
}

----- EmployeeController.java -----
package com.hrapp.controller;

import org.springframework.web.bind.annotation.*;
import java.util.*;
import com.hrapp.model.Employee;
import com.hrapp.service.EmployeeService;

@RestController
@RequestMapping("/employees")
public class EmployeeController {
    private EmployeeService service;

    public EmployeeController(EmployeeService service) {
        this.service = service;
    }

    @GetMapping
    public List<Employee> getAll() {
        return service.getAll();
    }

    @PostMapping
    public String add(@RequestBody Employee e) {
        service.add(e);
        return "Employee added successfully";
    }
}

----- HrAppApplication.java -----
package com.hrapp;

import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;

@SpringBootApplication
public class HrAppApplication {
    public static void main(String[] args) {
        SpringApplication.run(HrAppApplication.class, args);
    }
}

Q3. User Registration and Login Validation
------------------------------------------
Code:
----- User.java -----
package com.userapp.model;

public class User {
    private String username;
    private String password;
    private String email;

    public String getUsername() { return username; }
    public void setUsername(String username) { this.username = username; }
    public String getPassword() { return password; }
    public void setPassword(String password) { this.password = password; }
    public String getEmail() { return email; }
    public void setEmail(String email) { this.email = email; }
}

----- UserController.java -----
package com.userapp.controller;

import org.springframework.web.bind.annotation.*;
import com.userapp.model.User;

@RestController
@RequestMapping("/user")
public class UserController {
    @PostMapping("/register")
    public String register(@RequestBody User u) {
        if (u.getEmail().contains("@") && u.getPassword().length() > 5)
            return "User registered successfully";
        return "Invalid registration details";
    }

    @PostMapping("/login")
    public String login(@RequestBody User u) {
        if ("admin".equals(u.getUsername()) && "admin123".equals(u.getPassword()))
            return "Login successful";
        return "Invalid credentials";
    }
}

Q1. ERP System using Spring
---------------------------
Code:
----- ERPController.java -----
package com.erpapp.controller;

import org.springframework.web.bind.annotation.*;
import java.util.*;

@RestController
@RequestMapping("/erp")
public class ERPController {
    private List<String> inventory = new ArrayList<>();
    private List<String> sales = new ArrayList<>();

    @PostMapping("/addInventory")
    public String addInventory(@RequestBody String item) {
        inventory.add(item);
        return "Item added to inventory";
    }

    @GetMapping("/inventory")
    public List<String> getInventory() { return inventory; }

    @PostMapping("/sale")
    public String addSale(@RequestBody String item) {
        sales.add(item);
        return "Sale recorded";
    }

    @GetMapping("/sales")
    public List<String> getSales() { return sales; }
}

HOW TO RUN THE FILES
--------------------
1. Install Java 17+ and Spring Boot CLI or use Spring Initializr (https://start.spring.io/).
2. Create a new Spring Boot project in your IDE (Eclipse or IntelliJ).
3. Add dependencies:
   - Spring Web
   - Spring Boot DevTools
4. Add .java files under src/main/java/com/<projectname>/.
5. Run the main application class (e.g., HrAppApplication.java).
6. Test endpoints with browser or Postman:
   - GET http://localhost:8080/employees
   - POST http://localhost:8080/user/register
   - GET http://localhost:8080/inventory
